# ---------------- Password Validator ----------------

def check_password(password):

    if len(password) < 8:
        return False

    upper = False
    lower = False
    digit = False

    for ch in password:
        if ch.isupper():
            upper = True
        elif ch.islower():
            lower = True
        elif ch.isdigit():
            digit = True

    if upper and lower and digit:
        return True
    else:
        return False


while True:

    password = input("Enter Password: ")

    if check_password(password):
        print("Password Accepted")
        break
    else:
        print("Invalid Password")
